Quick Start

Tích hợp API trong vài phút.

cURL
Python
JavaScript
# 1. Đăng ký lấy API key
curl -X POST http://localhost:8000/developers/register \
  -H "Content-Type: application/json" \
  -d '{"name":"Nguyen Van A","email":"dev@example.com","plan":"free"}'

# 2. Gọi API với key
curl -H "X-API-Key: sk_your_key_here" \
  http://localhost:8000/api/v1/products

# 3. Test sandbox (không cần key)
curl http://localhost:8000/sandbox/products
import httpx

# Đăng ký
resp = httpx.post("http://localhost:8000/developers/register", json={
    "name": "Nguyen Van A", "email": "dev@example.com", "plan": "free"
})
api_key = resp.json()["api_key"]

# Gọi API
resp = httpx.get(
    "http://localhost:8000/api/v1/products",
    headers={"X-API-Key": api_key}
)
print(resp.json())

# Xem quota
print("Remaining:", resp.headers["X-Quota-Remaining"])
// Đăng ký
const { api_key } = await fetch("http://localhost:8000/developers/register", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ name: "Dev", email: "dev@example.com", plan: "free" })
}).then(r => r.json());

// Gọi API
const { data } = await fetch("http://localhost:8000/api/v1/products", {
  headers: { "X-API-Key": api_key }
}).then(r => r.json());

console.log(data);
